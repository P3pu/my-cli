<h1 align="center">Task Tracker CLI</h1>

<p>
    <strong>COMANDOS</strong>:
    // adicionar
     task add "tarefa"
    // atualizar 
     task update {id} "tarefa"
    // deletar 
     task delete {id}
    // progresso da terefa
      task mark-in-progress {id} // progresso = in-progress
      task mark-done {id} // progresso = done
</p>
