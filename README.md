<h1 align="center">Task Tracker CLI</h1>

<p>
    <strong>COMANDOS</strong>: <br>
    // adicionar <br>
     task add "tarefa" <br>
    // atualizar <br>
     task update {id} "tarefa" <br>
    // deletar <br>
     task delete {id} <br>
    // progresso da terefa<br>
      task mark-in-progress {id} // progresso = in-progress <br>
      task mark-done {id} // progresso = done <br>
</p>
